  8.   const [todoEditing, setTodoEditing] = React.useState(null);
  9.   const [editingText, setEditingText] = React.useState("");
  11.   React.useEffect(() => {
  12.     const json = localStorage.getItem("todos");
  13.     const loadedTodos = JSON.parse(json);
  14.     if (loadedTodos) {
  15.       setTodos(loadedTodos);
  16.     }
  17.   }, []);
  19.   React.useEffect(() => {
  20.     if (todos.length > 0) {
  21.         const json = JSON.stringify(todos);
  22.         localStorage.setItem("todos", json);
  23.     }
  24.   }, [todos]);
  26.   function handleSubmit(e) {
  27.     e.preventDefault();
  29.     const newTodo = {
  30.       id: new Date().getTime(),
  31.       text: todo.trim(),
  32.       completed: false,
  33.     };
  34.     if (newTodo.text.length > 0 ) {
  35.         setTodos([...todos].concat(newTodo));
  36.         setTodo("");
  38.     } else {
  40.         alert("Enter Valid Task");
  41.         setTodo("");
  42.     }
  43.   }
  44.   function deleteTodo(id) {
  45.     let updatedTodos = [...todos].filter((todo) => todo.id !== id);
  46.     setTodos(updatedTodos);
  47.   }
  49.   function toggleComplete(id) {
  50.     let updatedTodos = [...todos].map((todo) => {
  51.       if (todo.id === id) {
  52.         todo.completed = !todo.completed;
  53.       }
  54.       return todo;
  55.     });
  56.     setTodos(updatedTodos);
  57.   }
  59.   function submitEdits(id) {
  60.     const updatedTodos = [...todos].map((todo) => {
  61.       if (todo.id === id) {
  62.         todo.text = editingText;
  63.         }
  64.         return todo;
  65.       });
  66.       setTodos(updatedTodos);
  67.       setTodoEditing(null);
  68.     }
  70.     return (
  71.         <div id="todo-list">
  72.           <h1>Todo List</h1>
  73.           <form onSubmit={handleSubmit}>
  74.             <input
  75.               type="text"
  76.               onChange={(e) => setTodo(e.target.value)}
  77.               value={todo}
  78.             />
  79.             <button type="submit">Add Todo</button>

  85.                   type="checkbox"
  86.                   id="completed"
  87.                   checked={todo.completed}
  88.                   onChange={() => toggleComplete(todo.id)}
  89.                 />
  90.                 {todo.id === todoEditing ? (
  91.                   <input
  92.                     type="text"
  93.                     onChange={(e) => setEditingText(e.target.value)}

  98.               </div>

  106.                 <button onClick={() => deleteTodo(todo.id)}>Delete</button>
  107.               </div>
  108.             </div>
  109.           ))}
  110.         </div>
  111.       );
  112.     };
